    GolangеӯҰд№ з¬”и®°-ж•°жҚ®зұ»еһӢ

    [toc]

    ж•ҙеһӢ

    ж•ҙеһӢеҲҶдёәдёӨзұ»пјҡжңүз¬ҰеҸ·ж•ҙеһӢ е’Ң ж— з¬ҰеҸ·ж•ҙеһӢ

    еңЁеҶ…еӯҳдёӯз”ұдёӨйғЁеҲҶиЎЁзӨә: {з¬ҰеҸ·дҪҚ}{ж•°еӯ—дҪҚзҪ®}

    жңүз¬ҰеҸ·ж•ҙеһӢ

    зұ»еһӢ й•ҝеәҰпјҲеӯ—иҠӮж•°пјү еҖјзҡ„иҢғеӣҙ
    int еңЁ32дҪҚе№іеҸ°дёә4пјҢеңЁ64дҪҚе№іеҸ°дёә8 еңЁ32дҪҚе№іеҸ°зӯүеҗҢдәҺint32пјҢеңЁ64дҪҚе№іеҸ°зӯүеҗҢдәҺint64
    int8 1 -128 ~127
    int16 2 -32768~32767
    int32 4 -2147483648~2147483647
    int64 8 -9223372036854775808~9223372036854775807

    ж— з¬ҰеҸ·ж•ҙеһӢ

    зұ»еһӢ й•ҝеәҰпјҲеӯ—иҠӮж•°пјү еҖјзҡ„иҢғеӣҙ
    uint еңЁ32дҪҚе№іеҸ°дёә4пјҢеңЁ64дҪҚе№іеҸ°дёә8 еңЁ32дҪҚе№іеҸ°зӯүеҗҢдәҺuint32пјҢеңЁ64дҪҚе№іеҸ°зӯүеҗҢдәҺuint64
    uint8 1 0~255 (2^8-1)
    uint16 2 0~65535 (2^16-1)
    uint32 4 0~4294967295 (2^32-1)
    uint64 8 0~18446744073709551615 (2^64-1)

    жіЁж„ҸпјҡдёҚеҗҢзұ»еһӢзҡ„ж•ҙеһӢж— жі•дә’зӣёиөӢеҖјпјҢйңҖиҰҒеҒҡзұ»еһӢиҪ¬жҚўгҖӮе…¶дёӯдҪҚж•°й•ҝзҡ„зұ»еһӢиҪ¬жҚўдёәдҪҚж•°зҹӯзҡ„зұ»еһӢпјҢжҲ–иҖ…ж— з¬ҰеҸ·зҡ„зұ»еһӢиҪ¬жҚўдёәжңүз¬ҰеҸ·зҡ„зұ»еһӢж—¶пјҢдјҡдёўеӨұеҮҶзЎ®жҖ§

    ж —еӯҗ

    var age uint8 
age = 18

var money int
money = -1000

    жө®зӮ№еһӢ

    жө®зӮ№еһӢиЎЁзӨәеёҰжңүе°Ҹж•°дҪҚзҡ„ж•°еӯ—пјҢд№ҹеҸ«жө®зӮ№ж•°

    еңЁеҶ…еӯҳдёӯз”ұдёүйғЁеҲҶиЎЁзӨәпјҡ{з¬ҰеҸ·дҪҚ}{жҢҮж•°дҪҚ}{жңүж•Ҳж•°еӯ—дҪҚ}

    з¬ҰеҸ·дҪҚз”ЁдәҺиЎЁзӨәжө®зӮ№ж•°зҡ„жӯЈиҙҹжҖ§пјҢе®ғеҚ жҚ®дәҶжө®зӮ№ж•°зҡ„жңҖй«ҳдҪҚгҖӮ0 иЎЁзӨәжӯЈж•°пјҢ1 иЎЁзӨәиҙҹж•°

    жҢҮж•°дҪҚз”ЁдәҺиЎЁзӨәжө®зӮ№ж•°зҡ„ж•°йҮҸзә§пјҢе®ғйҖҡеёёжҳҜдёҖдёӘеёҰз¬ҰеҸ·зҡ„ж•ҙж•°гҖӮжҢҮж•°йғЁеҲҶзҡ„й•ҝеәҰе’Ңжө®зӮ№ж•°зҡ„зұ»еһӢжңүе…іпјҢйҖҡеёёдёә 8 дҪҚжҲ– 11 дҪҚгҖӮжҢҮж•°йғЁеҲҶзҡ„еҖјеҸҜд»ҘдёәжӯЈж•°гҖҒиҙҹж•°жҲ–йӣ¶гҖӮ

    жңүж•Ҳж•°еӯ—дҪҚз”ЁдәҺиЎЁзӨәжө®зӮ№ж•°зҡ„зІҫеәҰгҖӮе®ғйҖҡеёёжҳҜдёҖдёӘеёҰз¬ҰеҸ·зҡ„е°Ҹж•°пјҢй•ҝеәҰд№ҹдёҺжө®зӮ№ж•°зҡ„зұ»еһӢжңүе…ігҖӮе°ҫж•°йғЁеҲҶзҡ„еҖјйҖҡеёёжҳҜдёҖдёӘе°Ҹж•°пјҢеҸҜд»ҘжҳҜжӯЈж•°гҖҒиҙҹж•°жҲ–йӣ¶гҖӮ

    зұ»еһӢ й•ҝеәҰпјҲеӯ—иҠӮж•°пјү жҢҮж•°дҪҚ(bit) жңүж•Ҳж•°еӯ—дҪҚ(bit)
    float32 4 8 23
    float64 8 11 52

    ж —еӯҗ

    var speed float32
speed = 35.75
height := 1.78

    жіЁж„Ҹпјҡ

    1. еңЁдёҚеЈ°жҳҺжө®зӮ№ж•°зұ»еһӢзҡ„жғ…еҶөдёӢпјҢй»ҳи®ӨдјҡиҜҶеҲ«дёәfloat64зұ»еһӢ
    2. жө®зӮ№еһӢдёҚжҳҜз»қеҜ№зІҫзЎ®зҡ„ж•°еҖјзұ»еһӢпјҢеӯҳеңЁдёҖе®ҡзҡ„иҜҜе·®пјҢеӣ жӯӨеңЁиҝӣиЎҢжө®зӮ№ж•°зҡ„жҜ”иҫғж—¶пјҢйңҖиҰҒжҢҮе®ҡзІҫеәҰ

    еёғе°”еһӢ

    еёғе°”зұ»еһӢжңүдёӨдёӘеҖј: true е’Ң falseпјҢеҸӘеҚ з”ЁдёҖдёӘbitзҡ„еӨ§е°ҸпјҢйҖҡеёёз”ЁдәҺжқЎд»¶еҲӨж–ӯ

    ж —еӯҗ

    var answer bool
answer = true

    еӯ—з¬ҰеһӢ

    еӯ—з¬ҰеһӢжңүдёӨз§Қпјҡbyte е’Ң runeпјҢдҪҝз”ЁдёӨдёӘиӢұж–ҮеҚ•еј•еҸ·иЎЁзӨә''

    зұ»еһӢ й•ҝеәҰ зј–з Ғ
    byte 1 UTF-8
    rune 4 Unicode

    ж —еӯҗ

    var name rune = вҖҷjruingвҖҳ
var name byte = 'jruing'

    еӯ—з¬ҰдёІеһӢ

    еӯ—з¬ҰдёІзұ»еһӢе…ій”®еӯ—stringпјҢдҪҝз”Ёз¬ҰеҸ·""иЎЁзӨәеӯ—з¬ҰдёІеӯ—йқўеёёйҮҸ

    ж —еӯҗ

    var name string
name = "Jruing"

    жіЁж„Ҹ

    1. еӯ—з¬ҰдёІзұ»еһӢж— жі•дҝ®ж”№еӯ—з¬ҰдёІдёӯзҡ„жҹҗдёҖдҪҚеӯ—з¬ҰпјҢеҸӘиғҪйҮҚж–°иөӢеҖј

    2. еӯ—з¬ҰдёІй»ҳи®Өзј–з ҒдёәUTF-8

    3. еҰӮжһңйңҖиҰҒж”№еҸҳеӯ—з¬ҰдёІдёӯзҡ„жҹҗдёҖдҪҚеӯ—з¬ҰпјҢйңҖиҰҒе…Ҳе°Ҷеӯ—з¬ҰдёІе…ҲиҪ¬жҚўдёә[]byteжҲ–[]runeпјҢж”№еҸҳж•°з»„дёӯеҜ№еә”дёӢж Үзҡ„е…ғзҙ пјҢеҶҚиҪ¬жҚўдёәдёҖдёӘж–°зҡ„еӯ—з¬ҰдёІ

    package main

import (
    "fmt"
    "strconv"
)

func main() {
    var v1 bool
    // е°Ҷеӯ—з¬ҰдёІиҪ¬жҚўдёәеёғе°”еһӢ
    v1, _ = strconv.ParseBool("true")
    fmt.Println(v1)

    var v2 float64
    // е°Ҷеӯ—з¬ҰдёІиҪ¬жҚўдёәжө®зӮ№еһӢпјҢз¬¬дәҢдҪҚеҸӮж•°дёәжө®зӮ№еһӢж•°зҡ„дҪҚж•°
    v2, _ = strconv.ParseFloat("2131.393", 64)
    fmt.Println(v2)

    var v3 int64
    // е°Ҷеӯ—з¬ҰдёІжҚўз®—дёәж•ҙеһӢпјҢз¬¬дәҢдҪҚеҸӮж•°иЎЁзӨәд»ҘеӨҡе°‘дҪҚиҝӣеҲ¶иҝӣиЎҢжҚўз®—пјҢз¬¬дёүдҪҚдёәжҚўз®—еҗҺзҡ„ж•°еҖјдҪҚж•°
    v3, _ = strconv.ParseInt("-116", 10, 64)
    fmt.Println(v3)

    var v4 uint64
    // е°Ҷеӯ—з¬ҰдёІжҚўз®—дёәж— з¬ҰеҸ·ж•ҙеһӢпјҢз¬¬дәҢдҪҚеҸӮж•°иЎЁзӨәд»ҘеӨҡе°‘дҪҚиҝӣеҲ¶иҝӣиЎҢжҚўз®—пјҢз¬¬дёүдҪҚдёәжҚўз®—еҗҺзҡ„ж•°еҖјдҪҚж•°
    v4, _ = strconv.ParseUint("23", 10, 64)
    fmt.Println(v4)

    var v5 string
    // е°Ҷеёғе°”еһӢиҪ¬жҚўдёәеӯ—з¬ҰдёІ
    v5 = strconv.FormatBool(false)
    fmt.Println(v5)

    var v6 string
    // е°Ҷжө®зӮ№еһӢиҪ¬жҚўдёәеӯ—з¬ҰдёІпјҢз¬¬дәҢдҪҚеҸӮж•°иЎЁзӨәиҫ“еҮәж јејҸпјҢ'f'дёәжҷ®йҖҡжҳҫзӨәж–№ејҸпјҢеҸ–е°Ҹж•°зӮ№еҗҺзҡ„дҪҚж•°еҒҡдёәзІҫзЎ®дҪҚж•°пјҢ'e'дёҺ'E'дёәжҢҮж•°еҪўејҸпјҢ'g'дёҺ'G'д»Ҙжңүж•Ҳж•°еӯ—дҪҚж•°еҒҡдёәзІҫзЎ®дҪҚж•°пјҢз¬¬дёүдҪҚиЎЁзӨәж•°еҖјзІҫзЎ®дҪҚж•°пјҢз¬¬еӣӣдҪҚиЎЁзӨәж•°еҖјжүҖеҚ з©әй—ҙдҪҚж•°
    v6 = strconv.FormatFloat(3.141592679, 'f', 6, 64)
    fmt.Println(v6)

    var v7 string
    // е°Ҷеёғе°”еһӢиҪ¬жҚўдёәж•ҙеһӢ
    v7 = strconv.FormatInt(-667434, 10)
    fmt.Println(v7)

    var v8 string
    // е°Ҷеёғе°”еһӢиҪ¬жҚўдёәж•ҙеһӢ
    v8 = strconv.FormatUint(32434, 10)
    fmt.Println(v8)

    var v9 int
    // е°Ҷеӯ—з¬ҰдёІ"2"иҪ¬жҚўдёәж•ҙеһӢ
    v9, _ = strconv.Atoi("2")
    fmt.Println(v9)

    var v10 string
    // е°Ҷж•ҙеһӢеҖј3иҪ¬жҚўдёәеӯ—з¬ҰдёІ
    v10 = strconv.Itoa(3)
    fmt.Println(v10)
}

    ж•°з»„

    ж•°з»„иЎЁзӨәзҡ„жҳҜеҗҢдёҖз§Қж•°жҚ®зұ»еһӢзҡ„йӣҶеҗҲ,ж•°з»„дёӯзҡ„жҜҸдёҖдёӘе…ғзҙ з§°дёәж•°з»„е…ғзҙ пјҢжҜҸдёӘе…ғзҙ йғҪжңүеҜ№еә”зҡ„дёӢж Үең°еқҖпјҢд»Һ0ејҖе§ӢпјҢеңЁе®ҡд№үж•°з»„ж—¶йңҖиҰҒжҢҮе®ҡж•°з»„зҡ„й•ҝеәҰпјҢж•°з»„й•ҝеәҰеҝ…йЎ»жҳҜж•ҙеһӢеёёйҮҸ

    зјәзӮ№пјҡ

    1. ж•°з»„е…ғзҙ й•ҝеәҰеӣәе®ҡпјҢдёҚж”ҜжҢҒе…ғзҙ зҡ„еҲ йҷӨе’Ңж–°еўһ
    2. ж•°з»„иҝӣиЎҢеҸӮж•°дј йҖ’ж—¶йҮҮз”ЁеҖјдј йҖ’пјҢйңҖиҰҒиҝӣиЎҢе®Ңж•ҙиөӢеҖј

// иҺ·еҸ–ж•°з»„зҡ„й•ҝеәҰ len()

    package main

import (
    "fmt"
)

// е®ҡд№үдёҖдёӘRunnerжҺҘеҸЈпјҢе…¶дёӯеҢ…еҗ«Run()ж–№жі•
type Runner interface {
    Run()
}

// е®ҡд№үдёҖдёӘSwimmerжҺҘеҸЈпјҢе…¶дёӯеҢ…еҗ«Swim()ж–№жі•
type Swimmer interface {
    Swim
}

// е®ҡд№үдёҖдёӘCatзұ»еһӢ
type Cat struct{}

// Catзұ»еһӢе®һзҺ°дәҶRunж–№жі•пјҢиҜҙжҳҺзҢ«жҳҜRunner
func (c Cat) Run() {
    fmt.Println("cat can run.")
}

// е®ҡд№үдёҖдёӘDogзұ»еһӢ
type Dog struct{}

// Dogзұ»еһӢе®һзҺ°дәҶRunж–№жі•пјҢиҜҙжҳҺзӢ—жҳҜRunner
func (d Dog) Run() {
    fmt.Println("dog can run.")
}

// Dogзұ»еһӢе®һзҺ°дәҶSwimж–№жі•пјҢиҜҙжҳҺзӢ—жҳҜSwimmer
func (d Dog) Swim() {
    fmt.Println("dog can swim.")
}

func main() {
    // е®ҡд№үдёҖдёӘRunnerжҺҘеҸЈеҸҳйҮҸrunnerпјҢй»ҳи®ӨеҖјnil
    var runner Runner
    // е®ҡд№үдёҖдёӘSwimmerжҺҘеҸЈеҸҳйҮҸswimmerпјҢй»ҳи®ӨеҖјnil
    var swimmer Swimmer

    // е®һдҫӢеҢ–дёҖдёӘCatзұ»еһӢеҸҳйҮҸcat
    var cat Cat
    // е®һдҫӢеҢ–дёҖдёӘDogзұ»еһӢеҸҳйҮҸdog
    var dog Dog

    // е°ҶcatиөӢеҖјз»ҷжҺҘеҸЈrunner
    runner = cat
    // йҖҡиҝҮrunnerжү§иЎҢcatзҡ„Run()ж–№жі•
    runner.Run()

    // е°ҶdogиөӢеҖјз»ҷжҺҘеҸЈrunner
    runner = dog
    // йҖҡиҝҮrunnerжү§иЎҢdogзҡ„Run()ж–№жі•
    runner.Run()

    // е°ҶdogиөӢеҖјз»ҷжҺҘеҸЈswimmer
    swimmer = dog
    // йҖҡиҝҮswimmerжү§иЎҢdogзҡ„Swim()ж–№жі•
    swimmer.Swim()
}
